mindmap
  root(Electricity)
    (Electric Current)
      ::icon(fa:bolt)
      (Definition)
        (Flow of electric charge through a conductor)
        (Rate of flow of electric charges)
      (Formula)
        I = Q / t
      (Units)
        (Ampere (A))
          (1 A = 1 C/s)
        (Milliampere (mA))
          (1 mA = 10⁻³ A)
        (Microampere (µA))
          (1 µA = 10⁻⁶ A)
      (Charge (Q))
        (SI Unit: Coulomb (C))
        (1 C = charge of approx. 6 × 10¹⁸ electrons)
        (Charge of an electron = -1.6 × 10⁻¹⁹ C)
      (Time (t))
        (SI Unit: Second (s))
      (Conventional Current Direction)
        (Opposite to the direction of flow of electrons)
      (Measurement)
        (Ammeter)
          (Connected in series)
    (Electric Circuit)
      ::icon(fa:link)
      (Definition)
        (A continuous and closed path of an electric current)
      (Components)
        (Cell/Battery)
        (Electric Bulb)
        (Ammeter)
        (Plug Key/Switch)
      (Circuit Diagram Symbols)
        (Cell)
        (Battery)
        (Plug Key (open/closed))
        (Bulb)
        (Resistor)
        (Variable Resistor)
        (Ammeter)
        (Voltmeter)
    (Electric Potential Difference (V))
      ::icon(fa:battery-half)
      (Definition)
        (The difference of electric pressure along the conductor)
        (Work done to move a unit charge between two points)
      (Formula)
        V = W / Q
      (Units)
        (Volt (V))
          (1 V = 1 J/C)
      (Source of P.D.)
        (Battery/Cells)
          (Chemical action generates P.D.)
    (Ohm's Law)
      ::icon(fa:circle-nodes)
      (Statement)
        (At constant temperature, the electric current through a conductor is directly proportional to the potential difference across its ends.)
      (Formula)
        V ∝ I  => V = IR
      (Resistance (R))
        (Opposition to the flow of current)
        (SI Unit: Ohm (Ω))
        (Factors affecting Resistance)
          (Length (l))
            (R ∝ l)
          (Area of cross-section (A))
            (R ∝ 1/A)
          (Nature of material (Resistivity, ρ))
            (R = ρl/A)
          (Temperature)
    (Resistivity (ρ))
      ::icon(fa:wave-square)
      (Definition)
        (Intrinsic property of a material to resist the flow of electric current)
      (Units)
        (Ohm-meter (Ωm))
      (Relationship to Resistance)
        (R = ρl/A)
    (Series Combination of Resistors)
      ::icon(fa:stream)
      (Arrangement)
        (Resistors connected end-to-end)
      (Current)
        (Same through all resistors)
      (Potential Difference)
        (Divides among resistors)
        (V_total = V₁ + V₂ + V₃ + ...)
      (Equivalent Resistance)
        (R_eq = R₁ + R₂ + R₃ + ...)
    (Parallel Combination of Resistors)
      ::icon(fa:layer-group)
      (Arrangement)
        (Resistors connected across the same two points)
      (Current)
        (Divides among resistors)
        (I_total = I₁ + I₂ + I₃ + ...)
      (Potential Difference)
        (Same across all resistors)
      (Equivalent Resistance)
        (1/R_eq = 1/R₁ + 1/R₂ + 1/R₃ + ...)
    (Heating Effect of Electric Current)
      ::icon(fa:fire)
      (Joule's Law of Heating)
        (Heat produced (H) is proportional to)
          (Square of current (I²))
          (Resistance (R))
          (Time (t))
        (Formula)
          H = I²Rt
          (Also H = VIt, H = V²/R * t)
      (Applications)
        (Electric Heater)
        (Electric Iron)
        (Electric Kettle)
        (Electric Fuse)
          (Melts when current exceeds a certain limit)
    (Electric Power (P))
      ::icon(fa:lightbulb)
      (Definition)
        (Rate at which electric energy is consumed or supplied)
      (Formula)
        P = W / t
        P = VI
        P = I²R
        P = V²/R
      (Units)
        (Watt (W))
        (Kilowatt (kW))
          (1 kW = 1000 W)
      (Commercial Unit of Energy)
        (Kilowatt-hour (kWh))
          (1 kWh = 1 Unit of electricity)
          (1 kWh = 3.6 × 10⁶ J)
